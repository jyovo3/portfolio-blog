server {
    listen 80 default_server;
    listen [::]:80 default_server;

    root /root/portfolio-blog/out;
    index index.html;

    server_name _;

    location / {
        try_files $uri $uri/ /index.html;
    }

    location ~* \.(?:css|js|woff2?)$ {
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
}
